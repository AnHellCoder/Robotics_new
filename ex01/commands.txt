man - открыть мануал с описаниями команд.
cd - перейти по указанному пути.
mkdir - создать папку
rmdir - удалить папку
vim - открыть редактор кода
echo - напечатать текст
jot - генератор текста
time - распечатать время выполнения программы
rm - удалить файл
yes - вопсроизвести строку бесконечное количество раз.
